create table company(company_name VARCHAR(25), country VARCHAR(18), established_year VARCHAR(5), company_phone VARCHAR(15), primary key (company_name));
create table brands(brand_name VARCHAR(18), company_name VARCHAR(25), established_year VARCHAR(5), brand_phone VARCHAR(15), primary key (brand_name), foreign key (company_name) references company(company_name) on delete cascade);
create table dealers(dealer_name VARCHAR(18), d_address VARCHAR(10), dealer_phone VARCHAR(15), brand_name VARCHAR(18), primary key (dealer_name, d_address), foreign key (brand_name) references brands(brand_name) on delete cascade);
create table models(model_name VARCHAR(25), brand_name VARCHAR(18), body_style VARCHAR(18), default_price INTEGER, fuel VARCHAR(10), primary key (model_name), foreign key (brand_name) references brands(brand_name) on delete cascade);
create table options(model_name VARCHAR(25), color VARCHAR(10), engine VARCHAR(10), transmission VARCHAR(10), primary key (model_name, color, engine, transmission), foreign key (model_name) references models(model_name) on delete cascade);
create table suppliers(supplier_name VARCHAR(18), part_type VARCHAR(15), primary key (supplier_name, part_type));
create table manufacturing_plants(plant_name VARCHAR(18), company_name VARCHAR(25), primary key (plant_name), foreign key (company_name) references company(company_name) on delete cascade);
create table supply_for_sale(plant_name VARCHAR(18), supplier_name VARCHAR(18), part_type VARCHAR(15), manufacture_date DATE, primary key (plant_name, supplier_name, part_type, manufacture_date), foreign key (plant_name) references manufacturing_plants(plant_name) on delete cascade, foreign key (supplier_name, part_type) references suppliers(supplier_name, part_type) on delete cascade);
create table customers(customer_name VARCHAR(18), phone_number VARCHAR(15), c_address VARCHAR(18), annual_income INTEGER, primary key (customer_name, phone_number));
create table person_customers(customer_name VARCHAR(18), phone_number VARCHAR(15), gender VARCHAR(1), primary key (customer_name, phone_number), foreign key (customer_name, phone_number) references customers(customer_name, phone_number) on delete cascade);
create table company_customers(customer_name VARCHAR(18), phone_number VARCHAR(15), type VARCHAR(18), primary key (customer_name, phone_number), foreign key (customer_name, phone_number) references customers(customer_name, phone_number) on delete cascade);
create table vehicles(VIN VARCHAR(18), sale_price INTEGER, production_date DATE, on_sale VARCHAR(1), dealer_name VARCHAR(18), d_address VARCHAR(10), model_name VARCHAR(25), color VARCHAR(10), engine VARCHAR(10), transmission VARCHAR(10), plant_name VARCHAR(18), customer_name VARCHAR(18), phone_number VARCHAR(15), purchase_date DATE, primary key (VIN));
alter table vehicles add foreign key (dealer_name, d_address) references dealers(dealer_name, d_address) on delete cascade;
alter table vehicles add foreign key (model_name) references models(model_name) on delete cascade;
alter table vehicles add foreign key (model_name, color, engine, transmission) references options(model_name, color, engine, transmission) on delete cascade;
alter table vehicles add foreign key (plant_name) references manufacturing_plants(plant_name) on delete cascade;
alter table vehicles add foreign key (customer_name, phone_number) references customers(customer_name, phone_number) on delete cascade;
create table supply_parts(supplier_name VARCHAR(18), part_type VARCHAR(15), VIN VARCHAR(18), supply_date DATE, primary key (supplier_name, part_type, VIN), foreign key (supplier_name, part_type) references suppliers(supplier_name, part_type) on delete cascade, foreign key (VIN) references vehicles(VIN) on delete cascade);
